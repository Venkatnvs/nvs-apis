import requests
import json

reqUrl = "http://127.0.0.1:8000/api/zip/"

headersList = {
    "Accept": "*/*",
    "Authorization": "token YOUR_TOKEN",
    "Content-Type": "application/json" 
}

payload = json.dumps({
    "zip":"560078"
})

response = requests.request("POST", reqUrl, data=payload,  headers=headersList)

print(response.text)
